\begin{frame}{まとめ}
  \begin{itemize}
    \item 関数とは x の値に y がただ1つ対応する関係
    \item 定義域は x の範囲、値域は y の範囲
    \item 1次関数 y=ax+b のグラフは傾き a・切片 b の直線
    \item f(a+h) ≠ f(a)+h に注意
  \end{itemize}
