\begin{frame}{まとめ}
  \begin{itemize}
    \item y=a(x-p)²+q が標準形：軸 x=p、頂点 (p,q)
    \item a>0 で下に凸、a<0 で上に凸
    \item x 軸方向に p 平行移動 → x を x-p に置き換える
    \item y 軸方向に q 平行移動 → 右辺に +q
  \end{itemize}
