\begin{frame}{公式・性質}
  \begin{block}{}
  \begin{itemize}
    \item 平方完成の公式：$ax^2+bx+c = a\!\left(x+\dfrac{b}{2a}\right)^{\!2}-\dfrac{b^2-4ac}{4a}$
    \item 標準形の軸：$x=-\dfrac{b}{2a}$、頂点：$\left(-\dfrac{b}{2a},\,-\dfrac{b^2-4ac}{4a}\right)$
    \item 3点を通る放物線は $y=ax^2+bx+c$ として連立方程式を立てる
  \end{itemize}
  \end{block}
\end{frame}
