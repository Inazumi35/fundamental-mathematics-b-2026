  \begin{description}[labelwidth=5.5em]
    \item[\kw{判別式 $D$}] $ax^2+bx+c=0$ の解の判定量：$D=b^2-4ac$
    \item[\kw{$D>0$}] 放物線が $x$ 軸と\textbf{2点}で交わる（実数解2個）
    \item[\kw{$D=0$}] 放物線が $x$ 軸に\textbf{接する}（重解1個）
    \item[\kw{$D<0$}] 放物線が $x$ 軸と\textbf{交わらない}（実数解なし）
  \end{description}
  \end{block}
