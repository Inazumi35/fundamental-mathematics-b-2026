  \begin{block}{用語・定義}
  \begin{description}[labelwidth=5.5em]
    \item[\kw{分数関数}] $y=\dfrac{a}{x}$ や $y=\dfrac{ax+b}{cx+d}$ のような有理式で表される関数
    \item[\kw{双曲線}] $y=\dfrac{a}{x}$ のグラフの形（2本の曲線）
    \item[\kw{漸近線}] グラフが限りなく近づく直線（$y=\dfrac{a}{x}$ では $x$ 軸と $y$ 軸）
  \end{description}
  \end{block}
\end{frame}

% -----------------------------------------------
% スライド2: 公式・性質
% -----------------------------------------------
\begin{frame}{公式・性質}
  \begin{block}{}
  \begin{itemize}
    \item $y=\dfrac{a}{x}$：定義域 $x\neq 0$、値域 $y\neq 0$、漸近線 $x=0$・$y=0$
    \item $a>0$ のとき第1・第3象限、$a<0$ のとき第2・第4象限に分布
    \item $y=\dfrac{a}{x-p}+q$：漸近線 $x=p$ と $y=q$ に移動
  \end{itemize}
  \end{block}
\end{frame}

\begin{frame}{まとめ}
  \begin{itemize}
    \item $y=\dfrac{a}{x}$ のグラフは双曲線、漸近線は $x=0$ と $y=0$
    \item $y=\dfrac{a}{x-p}+q$ は漸近線が $x=p$ と $y=q$ に移動する
    \item 定義域・値域は漸近線の「除外点」から読み取る
  \end{itemize}
