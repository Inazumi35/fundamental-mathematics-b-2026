  \begin{itemize}

    \item $y=\dfrac{a}{x-p}+q$：漸近線 $x=p$、$y=q$
    \item $y=\sqrt{x-p}+q$：始点 $(p,\,q)$、右上がりの曲線
    \item $y=f^{-1}(x)$：$x$ と $y$ を入れ替えて $y$ について解く
  \end{itemize}
