\begin{frame}{重要公式一覧}
  \begin{block}{}
  \begin{itemize}
    \item 平方完成：$y=a(x-p)^2+q$、軸 $x=p$、頂点 $(p,\,q)$
    \item 判別式：$D=b^2-4ac$（$D>0$:2交点, $D=0$:接, $D<0$:交わらない）
    \item $y=\dfrac{a}{x-p}+q$：漸近線 $x=p$, $y=q$
    \item $y=\sqrt{x-p}+q$：始点 $(p,\,q)$、定義域 $x\geq p$
    \item 逆関数：$x$ と $y$ を入れ替えて解く
  \end{itemize}
