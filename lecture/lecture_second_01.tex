  \textbf{目標}：累乗根の定義と性質を理解し計算できる}

  \begin{block}{用語・定義}
  \begin{description}[labelwidth=5.5em]
    \item[\kw{n乗根}] $a^n = b$ を満たす数 $a$（$\sqrt[n]{b}$ と書く）
    \item[\kw{n乗根の定義（aのn乗根）}] $n$ が奇数のとき実数解1個、偶数のとき正の数のみ（$a > 0$）
    \item[\kw{$\sqrt[n]{a}$ の表記}] $n$ 乗根のうち正のもの（$n$ が偶数）または実数のもの
    \item[\kw{奇数乗根と偶数乗根の違い}] 奇数乗根は負の数でも定義できる；偶数乗根は $a \geq 0$ が必要
  \end{description}
  \end{block}
\end{frame}

% -----------------------------------------------
% スライド2: 公式・性質
% -----------------------------------------------
\begin{frame}{公式・性質}
  \begin{block}{}
  \begin{itemize}
    \item $(\sqrt[n]{a})^n = a$
    \item $\sqrt[n]{a^n} = a$（$n$ が奇数）
    \item $\sqrt[n]{a^n} = |a|$（$n$ が偶数）
    \item $\sqrt[n]{a} \cdot \sqrt[n]{b} = \sqrt[n]{ab}$
    \item $\sqrt[n]{a} \div \sqrt[n]{b} = \sqrt[n]{a/b}$
  \end{itemize}
  \end{block}
\end{frame}

\begin{frame}{まとめ}
  \begin{itemize}

    \item $(\sqrt[n]{a})^n = a$、$\sqrt[n]{a^n} = |a|$（$n$ が偶数）
