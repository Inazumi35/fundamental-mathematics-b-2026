    \item[\kw{指数関数：$y = a^x$（$a > 0$, $a \neq 1$）}] $y = a^x$（$a > 0$, $a \neq 1$）の形の関数
    \item[\kw{単調増加（$a > 1$）・単調減少（$0 < a < 1$）}] $a > 1$ のとき $x$ が増えると $y$ も増える；$0 < a < 1$ のとき $x$ が増えると $y$ が減る
    \item[\kw{漸近線：$y = 0$（$x$ 軸）}] $y = 0$（$x$ 軸）に限りなく近づくが交わらない
  \end{description}

\begin{frame}{まとめ}
  \begin{itemize}
    \item $y = a^x$ のグラフは常に点 $(0, 1)$ を通り，$y$ 軸の左右で単調増減
    \item $a > 1$ で単調増加，$0 < a < 1$ で単調減少
    \item 漸近線 $y = 0$，定義域は全実数，値域は $y > 0$
  \end{itemize}
