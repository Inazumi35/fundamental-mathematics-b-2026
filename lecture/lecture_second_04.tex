    \item[\kw{底が1より大：不等号の向きはそのまま}] $a > 1$ のとき $a^x > a^k \Rightarrow x > k$
    \item[\kw{底が1より小：不等号の向きが逆転}] $0 < a < 1$ のとき $a^x > a^k \Rightarrow x < k$
  \end{description}
