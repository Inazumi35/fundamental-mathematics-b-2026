  \begin{block}{用語・定義}
  \begin{description}[labelwidth=5.5em]
    \item[\kw{底の変換公式：$\log_a b = \dfrac{\log_c b}{\log_c a}$}] 任意の底 $c > 0$, $c \neq 1$ に変換できる
    \item[\kw{自然対数・常用対数（参考）}] 底が $e \approx 2.718$ の対数 $\ln x = \log_e x$（自然対数）；底が $10$ の対数 $\log_{10} x$（常用対数）
  \end{description}

\begin{frame}{まとめ}
  \begin{itemize}
    \item 底の変換公式 $\log_a b = \dfrac{\log_c b}{\log_c a}$ で底を統一する
    \item 対数の性質：$n \log_a b = \log_a b^n$，$\log_{a^n} b = \dfrac{1}{n}\log_a b$
    \item 底の変換で $\log_a b \cdot \log_b a = 1$ が成り立つ
  \end{itemize}
