  \begin{block}{用語・定義}
  \begin{description}[labelwidth=5.5em]
    \item[\kw{対数関数：$y = \log_a x$}] $y = \log_a x$（$a > 0$, $a \neq 1$, $x > 0$）の形の関数
    \item[\kw{単調増加（$a > 1$）・単調減少（$0 < a < 1$）}] 底 $a > 1$ のとき $x$ が増えると $y$ も増える；$0 < a < 1$ のとき $x$ が増えると $y$ が減る
    \item[\kw{漸近線：$x = 0$（$y$ 軸）}] $y$ 軸に限りなく近づくが交わらない；定義域は $x > 0$
    \item[\kw{真数条件の重要性}] $\log_a f(x)$ には $f(x) > 0$ が必要
  \end{description}
  \end{block}
\end{frame}
