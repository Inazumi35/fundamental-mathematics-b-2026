\begin{frame}{まとめ}
  \begin{itemize}
    \item $\sin^2\theta + \cos^2\theta = 1$，$\tan\theta = \sin\theta/\cos\theta$
    \item 特殊角：$\sin 30° = 1/2$，$\cos 45° = 1/\sqrt{2}$，$\tan 60° = \sqrt{3}$
    \item 相互関係 $1 + \tan^2\theta = 1/\cos^2\theta$ も確認する
  \end{itemize}
