  \textbf{目標}：鈍角（$90°$〜$180°$）の三角比を求め、相互関係を利用できる}

  \begin{block}{用語・定義}
  \begin{description}[labelwidth=5.5em]
    \item[\kw{鈍角（$90° < \theta < 180°$）の三角比}] $90° < \theta < 180°$ での $\sin\theta$, $\cos\theta$, $\tan\theta$ を単位円で定義
    \item[\kw{$\sin(180° - \theta) = \sin\theta$}] 補角の正弦は等しい
    \item[\kw{$\cos(180° - \theta) = -\cos\theta$}] 補角の余弦は符号が逆
    \item[\kw{$\tan(180° - \theta) = -\tan\theta$}] 補角の正接は符号が逆
  \end{description}
  \end{block}
\end{frame}

% -----------------------------------------------
% スライド2: 公式・性質
% -----------------------------------------------
\begin{frame}{公式・性質}
  \begin{block}{}
  \begin{itemize}
    \item 単位円を使った定義
    \item 第2象限での符号（$\sin > 0$、$\cos < 0$、$\tan < 0$）
    \item 相互関係：$\sin^2\theta + \cos^2\theta = 1$、$\tan\theta = \sin\theta/\cos\theta$
  \end{itemize}
  \end{block}
\end{frame}

% まとめ

\begin{frame}{まとめ}
  \begin{itemize}
    \item 鈍角の三角比は単位円で定義：$\cos\theta < 0$，$\sin\theta > 0$（$90° < \theta < 180°$）
    \item 補角の公式：$\sin(180° - \theta) = \sin\theta$，$\cos(180° - \theta) = -\cos\theta$
    \item $\sin^2\theta + \cos^2\theta = 1$ は鈍角でも成立する
  \end{itemize}
