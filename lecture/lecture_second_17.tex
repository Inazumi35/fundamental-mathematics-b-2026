    \item[\kw{$\sin(\pi + \theta) = -\sin\theta$ など（$\pi + \theta$ 型）}] $\sin(\pi+\theta) = -\sin\theta$，$\cos(\pi+\theta) = -\cos\theta$，$\tan(\pi+\theta) = \tan\theta$
    \item[\kw{$\sin(-\theta) = -\sin\theta$ など（$-\theta$ 型）}] $\sin(-\theta) = -\sin\theta$，$\cos(-\theta) = \cos\theta$，$\tan(-\theta) = -\tan\theta$
