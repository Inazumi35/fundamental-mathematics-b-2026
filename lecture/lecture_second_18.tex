  \begin{block}{用語・定義}
  \begin{description}[labelwidth=5.5em]
    \item[\kw{振幅}] $y = a\sin x$ のグラフの振れ幅 $|a|$
    \item[\kw{周期：$y = \sin x$ の周期は $2\pi$}] $y = \sin x$ の周期は $2\pi$，$y = \cos x$ も $2\pi$，$y = \tan x$ は $\pi$
    \item[\kw{平行移動：$y = \sin(x - p) + q$}] $y = \sin(x - p) + q$ でグラフを水平・垂直に移動する
  \end{description}
  \end{block}
