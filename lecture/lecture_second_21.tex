    \item[\kw{単位円・グラフを使った解法}] 単位円上で $y$ 座標（$\sin$）または $x$ 座標（$\cos$）が $k$ となる点を探す

  \end{description}

\begin{frame}{公式・性質}
  \begin{block}{}
  \begin{itemize}
    \item $\sin x = k$ の解：$x = \alpha$ と $x = \pi - \alpha$（$|k| \leq 1$）
    \item $\cos x = k$ の解：$x = \alpha$ と $x = 2\pi - \alpha$
    \item 不等式は範囲で答える
  \end{itemize}
  \end{block}
\end{frame}
