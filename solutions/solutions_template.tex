\[
y = (x - 1)^2 + 2
\]
}

\kotae{
頂点 $(1, 2)$、軸 $x = 1$

\begin{center}
\begin{tikzpicture}
\begin{axis}[
    axis lines=middle,
    xlabel={$x$},
    ylabel={$y$},
    xmin=-1, xmax=3,
    ymin=0, ymax=6,
    xtick={0,1,2},
    ytick={0,2},
    samples=100,
    width=\linewidth,
    height=0.75\linewidth
]
\addplot[blue, thick, domain=-0.5:2.5] {(x-1)^2 + 2};
\addplot[only marks, mark=*, red] coordinates {(1,2)};
\node[above right] at (axis cs:1,2) {$(1,2)$};
\addplot[dashed, gray] coordinates {(1,0) (1,6)};
\end{axis}
\end{tikzpicture}
\end{center}
